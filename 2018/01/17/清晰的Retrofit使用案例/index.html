//使用最普通的方式进行GET请求
@GET(“v2/movie/top250”)
Call getMovieMsg();

private void getMovieModelGet() {
    Retrofit retrofit = new Retrofit.Builder()
            .baseUrl("https://api.douban.com/")
            .addConverterFactory(GsonConverterFactory.create())
            .build();
    ApiStores apiStores = retrofit.create(ApiStores.class);
    Call<MovieModel> call = apiStores.getMovieMsg();
    call.enqueue(new Callback<MovieModel>() {
        @Override
        public void onResponse(Call<MovieModel> call, Response<MovieModel> response) {
            Logger.d(response.body().getTitle());
        }

        @Override
        public void onFailure(Call<MovieModel> call, Throwable t) {
            Logger.d(t.getMessage());
        }
    });
}

//使用Path进行GET请求
@GET(“v2/{path}/top250”)
Call getMoviePathMsg(@Path(“path”) String path);

private void getMoviePathGet() {
Retrofit retrofit = new Retrofit.Builder().baseUrl(“https://api.douban.com/“)
.addConverterFactory(GsonConverterFactory.create())
.build();
ApiStores apiStores = retrofit.create(ApiStores.class);
Call call = apiStores.getMoviePathMsg(“movie”);
call.enqueue(new Callback() {
@Override
public void onResponse(Call call, Response response) {
Logger.d(“通过path获取”+response.body().getTitle());
}

    @Override
    public void onFailure(Call<MovieModel> call, Throwable t) {
        Logger.d(t.getMessage());
    }
});

}
